//Procedure creation
CREATE OR REPLACE PROCEDURE P_AUTHENTICATION 
(
  USERNAME IN VARCHAR2 
, PASSWORD IN VARCHAR2 
, RESULT OUT VARCHAR2 
) AS 
CNT NUMBER(3);
BEGIN
  SELECT COUNT(*) INTO CNT FROM USERINFO WHERE UNAME=USERNAME AND PWD=PASSWORD;
  IF(CNT<>0) then
    RESULT:='VALID CREDENTIALS';
  ELSE
    RESULT:='INVALID CREDENTIALS';
  END IF;
    
END P_AUTHENTICATION;

//Table creation
CREATE TABLE "SYSTEM"."USERINFO" 
(	"UNAME" VARCHAR2(255 BYTE), 
	"PWD" VARCHAR2(255 BYTE)
)
